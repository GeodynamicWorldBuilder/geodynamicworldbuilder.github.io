57 TEST_CASE("WorldBuilder Point: Testing initialize and operators")
58 {

60  Point<2> p2;
61  Point<3> p3;
62 
63  REQUIRE(p2.get_array() == std::array<double,2> {0,0});
64  REQUIRE(p3.get_array() == std::array<double,3> {0,0,0});
65

67  const Point<3> p3_array(std::array<double,3> {1,2,3});
68 
69  REQUIRE(p2_array.get_array() == std::array<double,2> {1,2});
70  REQUIRE(p3_array.get_array() == std::array<double,3> {1,2,3});
71 
72  const Point<2> p2_point(p2_array);
73  const Point<3> p3_point(p3_array);
74 
75  REQUIRE(p2_point.get_array() == std::array<double,2> {1,2});
76  REQUIRE(p3_point.get_array() == std::array<double,3> {1,2,3});
77 
78  const Point<2> p2_explicit(3,4);
79  const Point<3> p3_explicit(4,5,6);
80 
81  REQUIRE(p2_explicit.get_array() == std::array<double,2> {3,4});
82  REQUIRE(p3_explicit.get_array() == std::array<double,3> {4,5,6});
83 
87  // Test assign operator

89  p3 = p3_array;

92  REQUIRE(p3.get_array() == std::array<double,3> {1,2,3});

96  p3 = 2 * p3 * 1.0;
97 
98  REQUIRE(p2.get_array() == std::array<double,2> {2,4});
99  REQUIRE(p3.get_array() == std::array<double,3> {2,4,6});

102  p3 *= 2;
103 
104  REQUIRE(p2.get_array() == std::array<double,2> {4,8});
105  REQUIRE(p3.get_array() == std::array<double,3> {4,8,12});

107  // Test dot operator

110 
111  // Test add operator

113  p3 = p3 + p3;
114 
115  REQUIRE(p2.get_array() == std::array<double,2> {8,16});

119  p3 += p3;
120 
121  REQUIRE(p2.get_array() == std::array<double,2> {16,32});
122  REQUIRE(p3.get_array() == std::array<double,3> {16,32,48});

135  REQUIRE(p3.get_array() == std::array<double,3> {4,8,12});

213 TEST_CASE("WorldBuilder Utilities: Point in polygon")
214 {
215  std::vector<Point<2> > point_list_4_elements(4);
216  point_list_4_elements[0] = Point<2>(0,0);
217  point_list_4_elements[1] = Point<2>(5,0);
218  point_list_4_elements[2] = Point<2>(5,5);
219  point_list_4_elements[3] = Point<2>(0,5);
220 
221  std::vector<Point<2> > point_list_3_elements(3);
222  point_list_3_elements[0] = Point<2>(10,10);
223  point_list_3_elements[1] = Point<2>(10,15);
224  point_list_3_elements[2] = Point<2>(15,15);
225 
226  std::vector<Point<2> > check_points(9);
227  check_points[0] = Point<2>(-1,-1);
228  check_points[1] = Point<2>(0,0);
229  check_points[2] = Point<2>(0,5);
230  check_points[3] = Point<2>(5,0);
231  check_points[4] = Point<2>(5,5);
232  check_points[5] = Point<2>(5,5.01);
233  check_points[6] = Point<2>(1,1);
234  check_points[7] = Point<2>(12.5,12);
235  check_points[8] = Point<2>(11.5,12);
236 
237  std::vector<std::array<bool,2> > awnsers(9);
238  awnsers[0] = {false,false};
239  awnsers[1] = {true,false};
240  awnsers[2] = {true,false};
241  awnsers[3] = {true,false};
242  awnsers[4] = {true,false};
243  awnsers[5] = {false,false};
244  awnsers[6] = {true,false};
245  awnsers[7] = {false,false};
246  awnsers[8] = {false,true};
247 
248  std::vector<std::array<double,2> > awnsers_signed_distance(9);
249  awnsers_signed_distance[0] = {-std::sqrt(2), -std::sqrt(11 * 11 + 11 * 11)};
250  awnsers_signed_distance[1] = {0,-std::sqrt(10 * 10 + 10 * 10)};
251  awnsers_signed_distance[2] = {0,-std::sqrt(125)};
252  awnsers_signed_distance[3] = {0,-std::sqrt(125)};
253  awnsers_signed_distance[4] = {0,-std::sqrt(50)};
254  awnsers_signed_distance[5] = {-std::sqrt(0.01 * 0.01),-std::sqrt(5 * 5 + 4.99 * 4.99)};
255  awnsers_signed_distance[6] = {1,-std::sqrt(9 * 9 + 9 * 9)};
256  awnsers_signed_distance[7] = {-10.2591422643,-0.3535533906};
257  awnsers_signed_distance[8] = {-9.5524865873,0.3535533906};
258 
259  for (unsigned int i = 0; i < check_points.size(); ++i)
260  {
261  INFO("checking point " << i << " = (" << check_points[i][0] << ":" << check_points[i][1] << ")");
262  REQUIRE(Utilities::polygon_contains_point(point_list_4_elements,check_points[i]) == awnsers[i][0]);
263  REQUIRE(Utilities::polygon_contains_point(point_list_3_elements,check_points[i]) == awnsers[i][1]);
264  REQUIRE(Utilities::signed_distance_to_polygon(point_list_4_elements,check_points[i]) == Approx(awnsers_signed_distance[i][0]));
265  REQUIRE(Utilities::signed_distance_to_polygon(point_list_3_elements,check_points[i]) == Approx(awnsers_signed_distance[i][1]));
266  }
267 
268  std::vector<Point<2> > point_list_2_elements(2);
269  REQUIRE_THROWS_WITH(Utilities::signed_distance_to_polygon(point_list_2_elements,check_points[0]),
270  Contains("Not enough polygon points were specified."));
271 
272  std::vector<Point<2> > point_list_1_elements(1);
273  REQUIRE_THROWS_WITH(Utilities::signed_distance_to_polygon(point_list_1_elements,check_points[0]),
274  Contains("Not enough polygon points were specified."));
275 
276  std::vector<Point<2> > point_list_0_elements(0);
277  REQUIRE_THROWS_WITH(Utilities::signed_distance_to_polygon(point_list_0_elements,check_points[0]),
278  Contains("Not enough polygon points were specified."));
